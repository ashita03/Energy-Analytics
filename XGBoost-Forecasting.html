def load_and_prepare_data(train_data, test_data):
    """
    Load and prepare the training and test datasets with proper NaN handling
    """
    # Create a copy to avoid modifying the original dataframe
    train_data = train_data.copy()
    test_data = test_data.copy()
    
    # Add time-based features
    for df in [train_data, test_data]:
        # Simple day of week approximation
        df['dayofweek'] = df['Day'] % 7
        
        # Create season (1: Winter, 2: Spring, 3: Summer, 4: Fall)
        df['season'] = pd.cut(df['Month'], 
                            bins=[0, 3, 6, 9, 12], 
                            labels=[1, 2, 3, 4],
                            include_lowest=True)
        df['season'] = df['season'].fillna(2).astype(int)  # Fill NaN with middle season
        
        # Create time of day (0: Night, 1: Morning, 2: Afternoon, 3: Evening)
        df['time_of_day'] = pd.cut(df['Hour'], 
                                  bins=[-1, 5, 11, 17, 23], 
                                  labels=[0, 1, 2, 3],
                                  include_lowest=True)
        df['time_of_day'] = df['time_of_day'].fillna(1).astype(int)  # Fill NaN with morning
        
        # Add month and hour directly
        df['month'] = df['Month']
        df['hour'] = df['Hour']
        
        # Fill NaN values in temperature and GHI columns with mean values
        temp_cols = [col for col in df.columns if 'Temp' in col]
        ghi_cols = [col for col in df.columns if 'GHI' in col]
        
        for col in temp_cols:
            df[col] = df[col].fillna(df[col].mean())
        
        for col in ghi_cols:
            df[col] = df[col].fillna(0)  # Fill GHI with 0 for missing values (likely nighttime)
    
    # Print data info after preparation
    print("\nTraining Data Info:")
    print(train_data.info())
    print("\nMissing values in training data:")
    print(train_data.isnull().sum())
    
    return train_data, test_data

def perform_eda(train_data):
    """
    Perform exploratory data analysis
    """
    print("\nExploratory Data Analysis:")
    print("\nBasic Statistics for Load:")
    print(train_data['Load'].describe())
    
    # Check for missing values
    print("\nMissing Values:")
    print(train_data.isnull().sum())
    
    # Correlation analysis
    numerical_cols = train_data.select_dtypes(include=[np.number]).columns
    correlation = train_data[numerical_cols].corr()
    
    # Create correlation heatmap
    plt.figure(figsize=(12, 8))
    sns.heatmap(correlation, annot=True, cmap='coolwarm', center=0)
    plt.title('Correlation Heatmap')
    plt.tight_layout()
    plt.show()
    
    # Load distribution
    plt.figure(figsize=(10, 6))
    sns.histplot(train_data['Load'], bins=50)
    plt.title('Load Distribution')
    plt.show()
    
    # Load by hour
    plt.figure(figsize=(10, 6))
    avg_load_by_hour = train_data.groupby('Hour')['Load'].mean()
    avg_load_by_hour.plot()
    plt.title('Average Load by Hour')
    plt.xlabel('Hour')
    plt.ylabel('Average Load')
    plt.show()
    
    # Load by season
    plt.figure(figsize=(10, 6))
    avg_load_by_season = train_data.groupby('season')['Load'].mean()
    avg_load_by_season.plot(kind='bar')
    plt.title('Average Load by Season')
    plt.xlabel('Season')
    plt.ylabel('Average Load')
    plt.show()
    
    return correlation

def prepare_features(data, current_row=None):
    """
    Prepare features for the model, ensuring we only use data from current or past dates
    """
    feature_cols = ['Site-1 Temp', 'Site-2 Temp', 'Site-3 Temp', 'Site-4 Temp', 'Site-5 Temp',
                   'Site-1 GHI', 'Site-2 GHI', 'Site-3 GHI', 'Site-4 GHI', 'Site-5 GHI',
                   'hour', 'dayofweek', 'month', 'season', 'time_of_day']
    
    if current_row is not None:
        # Filter data to only include current and past observations
        mask = ((data['Year'] < current_row['Year']) |
                ((data['Year'] == current_row['Year']) & (data['Month'] < current_row['Month'])) |
                ((data['Year'] == current_row['Year']) & (data['Month'] == current_row['Month']) & 
                 (data['Day'] < current_row['Day'])) |
                ((data['Year'] == current_row['Year']) & (data['Month'] == current_row['Month']) & 
                 (data['Day'] == current_row['Day']) & (data['Hour'] <= current_row['Hour'])))
        data = data[mask]
    
    return data[feature_cols]

def train_and_evaluate_model(train_data):
    """
    Train XGBoost model with GridSearch and evaluate performance
    """
    # Define parameter grid
    param_grid = {
        'max_depth': [3, 5, 7],
        'learning_rate': [0.01, 0.1, 0.3],
        'n_estimators': [100, 200, 300],
        'min_child_weight': [1, 3, 5]
    }
    
    # Initialize base model
    base_model = XGBRegressor(
        random_state=42,
        tree_method='hist'  # Using histogram-based algorithm for better performance
    )
    
    # Perform GridSearch
    grid_search = GridSearchCV(
        estimator=base_model,
        param_grid=param_grid,
        cv=5,
        scoring='neg_mean_squared_error',
        verbose=1,
        n_jobs=-1
    )
    
    # Prepare features and target
    X = prepare_features(train_data)
    y = train_data['Load']
    
    # Print feature types before fitting
    print("\nFeature dtypes:")
    print(X.dtypes)
    
    # Check for any remaining NaN values
    print("\nAny NaN values in features:")
    print(X.isnull().sum())
    
    # Fit GridSearch
    grid_search.fit(X, y)
    
    # Print results
    print("\nGrid Search Results:")
    print(f"Best parameters: {grid_search.best_params_}")
    print(f"Best score: {np.sqrt(-grid_search.best_score_)}")
    
    # Get detailed CV results
    cv_results = pd.DataFrame(grid_search.cv_results_)
    cv_results = cv_results.sort_values('rank_test_score')
    
    print("\nTop 5 parameter combinations:")
    for i in range(5):
        params = cv_results.iloc[i]['params']
        score = np.sqrt(-cv_results.iloc[i]['mean_test_score'])
        print(f"\nRank {i+1}:")
        print(f"Parameters: {params}")
        print(f"RMSE: {score:.2f}")
    
    return grid_search.best_estimator_

def make_predictions(model, test_data):
    """
    Make predictions for test data using rolling window approach
    """
    predictions = []
    
    # Sort test data by Year, Month, Day, Hour
    test_data = test_data.sort_values(['Year', 'Month', 'Day', 'Hour'])
    
    # Make predictions for each row in test set
    for idx, row in test_data.iterrows():
        # Prepare features using only data up to current date
        X = prepare_features(test_data, row)
        X_current = prepare_features(test_data.iloc[[idx]])  # Get features for current timestamp
        
        # Make prediction
        pred = model.predict(X_current)[0]
        predictions.append(pred)
    
    return predictions

def evaluate_predictions(y_true, y_pred):
    """
    Evaluate predictions only on non-missing values
    """
    # Convert inputs to numpy arrays if they aren't already
    y_true = np.array(y_true)
    y_pred = np.array(y_pred)
    
    # Check if shapes match
    if y_true.shape != y_pred.shape:
        print(f"Shape mismatch: y_true {y_true.shape}, y_pred {y_pred.shape}")
        return
    
    # Remove rows where actual values are missing
    mask = ~np.isnan(y_true)
    y_true_clean = y_true[mask]
    
    # Ensure mask is applied correctly to y_pred
    # Use boolean indexing instead of trying to use mask as an integer index
    y_pred_clean = np.array([y_pred[i] for i in range(len(y_pred)) if mask[i]])
    
    if len(y_true_clean) > 0:
        rmse = np.sqrt(mean_squared_error(y_true_clean, y_pred_clean))
        mae = mean_absolute_error(y_true_clean, y_pred_clean)
        r2 = r2_score(y_true_clean, y_pred_clean)
        
        print("\nFinal Model Performance (on non-missing values):")
        print(f"Number of samples evaluated: {len(y_true_clean)}")
        print(f"RMSE: {rmse:.2f}")
        print(f"MAE: {mae:.2f}")
        print(f"R2 Score: {r2:.2f}")
    else:
        print("\nNo non-missing values available for evaluation")
